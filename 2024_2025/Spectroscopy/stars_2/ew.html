def measure_ew(filename, i):
    hdul = fits.open(filename)
    data = hdul[0].data
    header = hdul[0].header

    crval = header['CRVAL1']
    cdelt = header['CDELT1']
    npix = len(data)

    wavelength = crval + np.arange(npix) * cdelt
    flux = data
    flux = np.asarray(flux, dtype=np.float64)

    ew_results = {}

    fig, ax = plt.subplots(1, 1, figsize=(12, 12))

    for value in line_wavelengths.values():
        ax.axvline(value, color='blue', linestyle='--')

    ax.plot(wavelength, flux, color='black')

    for line_name, line_wav in line_wavelengths.items():

        cont_range = continuum_points[i][line_name]
        
        # Get closest flux values to continuum points
        cont_flux_left = flux[np.argmin(np.abs(wavelength - cont_range[0]))]
        cont_flux_right = flux[np.argmin(np.abs(wavelength - cont_range[1]))]
        
        # Define linear continuum between two points
        cont_slope = (cont_flux_right - cont_flux_left) / (cont_range[1] - cont_range[0])
        cont_intercept = cont_flux_left - cont_slope * cont_range[0]
        
        # Select line region
        line_mask = (wavelength >= cont_range[0]) & (wavelength <= cont_range[1])
        line_wave = wavelength[line_mask]
        line_flux = flux[line_mask]
        
        # Compute continuum flux over the line region
        cont_flux_interp = cont_slope * line_wave + cont_intercept

        ax.plot(wavelength, flux, color='black')
        ax.plot(line_wave, cont_flux_interp, color='red')
        
        # Calculate EW
        ew = np.sum((1 - line_flux / cont_flux_interp) * (line_wave[1] - line_wave[0]))
        ew_results[line_name] = ew

    plt.show()

    return ew_results
